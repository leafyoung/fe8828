fun_mean <- function(x){
  mean(x) %>%
  data.frame(y=.,label=mean(x,na.rm=T))}
ggplot(bank, aes(poutcome, duration)) + geom_boxplot() + geom_point()+
   stat_summary(fun.y = mean,geom = "point", colour="blue", size=3) + 
  stat_summary(fun.data=fun_mean, geom="text", vjust= -0.5)
